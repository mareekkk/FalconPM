# Make sure map-server sees rAthena header files
target_include_directories(map-server PRIVATE
    ${RA_SRC}            # e.g. /home/marek/rathena/src
    ${RA_SRC}/map        # e.g. /home/marek/rathena/src/map
    ${RA_SRC}/common     # e.g. /home/marek/rathena/src/common
)

# Ensure bootstrap is included
target_sources(map-server PRIVATE
    peregrine.cpp
    falconpm_bootstrap.cpp
)

# Ensure symbols exported for plugin lookups
target_link_options(map-server PRIVATE -rdynamic)
